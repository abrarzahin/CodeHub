Add two num


class Solution {
public ListNode addTwoNumbers(ListNode l1, ListNode l2) {


int carry = 0;
ListNode head = null;
ListNode preNode = null;
ListNode temp = null;
int indx = 0;

while (l1!=null || l2!=null){
if(l1==null && l2!=null){
temp = new ListNode((l2.val + carry)%10);
carry = (l2.val + carry) > 9 ? 1:0 ;
}
if(l1!=null && l2==null){
temp = new ListNode((l1.val + carry)%10 );
carry = (l1.val + carry) > 9 ? 1:0 ;
}
if(l1!=null && l2!=null){
temp = new ListNode((l1.val + l2.val + carry)%10);
carry = (l1.val + l2.val + carry) > 9 ? 1:0 ;
}


if(l1!=null) {
l1 = l1.next;
}
if(l2!=null) {
l2 = l2.next;
}

if(indx == 0) {
head = temp;
preNode = head;
indx ++;
continue;
}

preNode.next = temp;
preNode = temp;
}
if(carry > 0) {
preNode.next = new ListNode(1);
}

return head;


}
}
