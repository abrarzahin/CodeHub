Binary Tree Inorder Traversal


Share
Given a binary tree, return the level order traversal of its nodes' values. (ie, from left to right, level by level).

For example:
Given binary tree [3,9,20,null,null,15,7],
3
/ \
9 20
/ \
15 7
return its level order traversal as:
[
[3],
[9,20],
[15,7]
] 

        
                    class Solution {

                // please ignore the . inside the variable type
                // these are  used to make <> visible otherwise html things that those are tags

                        public List <.List <.Integer>> levelOrder(TreeNode root) {
                            List <.List<.Integer>> result = new ArrayList<>(); // ignore . inside list and integer
                            Queue<.TreeNode> q = new LinkedList<>();       // same for treenode
                            if (root == null) {
                                return result;
                            }
                
                            q.add(root);         
                            while (!q.isEmpty()) {                 
                                Queue <.TreeNode> temp = new LinkedList<>(); // same for these 2 lines
                                List<.Integer> subList = new ArrayList<>();
                                
                                while (!q.isEmpty()) {
                                    TreeNode n = q.peek();
                                    
                                    subList.add(n.val);
                                    
                                    q.remove();
                                    
                                    if (n.left != null) {
                                        temp.add(n.left);
                                    }
                                    
                                    if (n.right != null) {
                                        temp.add(n.right);
                                    }
                                }
                                
                                q = temp;
                                
                                result.add(subList);
                            }
                            
                            return result;
                        }
                    }
